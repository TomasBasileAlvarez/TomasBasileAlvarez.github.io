# oscilator equation
# We solve d^2 x / dt^2 = -k x
import numpy as np
from matplotlib import pyplot as plt

# we define v = d x / dt 
# and the other equation is dv / dt = -k x
#this is our new system of equations

k = 10

def f(r,t):
    x = r[0]
    v = r[1]
    fx = v
    fv = -k*x
    return(np.array([fx,fv],float))

a = 0.0
b = 10.0
N = 1000
h = (b-a)/N


tpoints = np.arange(a,b,h)
xpoints = []
vpoints = []

#initial values
# x = 1, v = 0
r = np.array([1,0],float)
for t in tpoints:
    xpoints.append(r[0])
    vpoints.append(r[1])
    k1 = h*f(r,t)
    k2 = h*f(r+0.5*k1,t+0.5*h)
    k3 = h*f(r+0.5*k2,t+0.5*h)
    k4 = h*f(r+k3,t+h)
    r += (k1+2*k2+2*k3+k4)/6
plt.plot(tpoints,xpoints)
plt.plot(tpoints,vpoints)
plt.xlabel("t")
plt.show()
plt.plot(xpoints,vpoints)


plt.show()
