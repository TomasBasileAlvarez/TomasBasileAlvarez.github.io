# La función toma como entrada f, los límites a,b y la cantidad de divisiones N

def simp(f,a,b,N=2000):
    # calculamos h según la def
    h = (b-a)/N
    
    # Hacemos un conjunto de valores de x desde a hasta b, en los que evaluaremos f
    x = np.linspace(a,b,N+1)
    # Evaluamos f en todos estos puntos
    y = f(x)
    
    #calculamos el resultado según la fórmula.
    # Necesitamos sumar sobre los impares (la suma de y[1::2]) 
    # y sobre los pares y[0:N-1:2]
    res = h/3 * (f(a)+f(b)+4*np.sum(y[1::2])+2*np.sum(y[0:N-1:2]))
    return(res)

2. Periodo de un Péndulo¶

El periodo de un péndulo simple de longitud $L$ es

$$ \tau = 4h (\theta_0) \sqrt{\dfrac{L}{g}}$$

Donde $g$ es el valor de la gravedad, $\theta_0$ representa la amplitud angular y

$$ h(\theta_0) = \int_0^{\pi/2} \dfrac{d\theta}{\sqrt{1-\sin^2 \left( \dfrac{\theta_0}{2} \right) \sin^2 (\theta)}}$$

(i) Calcula el periodo de oscilación para diferentes ángulos $\theta_0 = \pi/12 , \pi/6, \pi/4$

Definimos la funcion h, que se calcula como una integral

# h toma el valor th0 (theta cero) y a partir de eso nos da un resultado según la integral
def h(th0):
    #primero definimos el integrando en la variable th usando una función lambda
    func = lambda th: 1/np.sqrt(1-(np.sin(th0/2))**2*(np.sin(th))**2)
    # res es el resultado de la integral, que calculamos usando simpson
    res = simp(func, 0, np.pi/2)
    return(res)

#Esta función calcula el periodo de la oscilación
#toma como entrada el th0 y L (predeterminado a 10)
def periodo(th0,L=10):
    # Calcula el valor de h correspondiente y luego usa la def. de tau
    tau = 4*h(th0)*np.sqrt(L/9.81)
    return(tau)

# Calculamos el periodo para distintos ángulos
