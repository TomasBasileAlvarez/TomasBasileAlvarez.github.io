#Function to create the matrix Zeta of dimensions NxT with correlatino c
def matZeta(N,c):
    mat = np.zeros((N,N))
    for i in range(N):
        for j in range(N):
            if i==j:
                mat[i][j]=1
            else:
                mat[i][j]=c
    return(mat)

#Create a matrix of size N,T with random normal entries with mean 0 and std 1
def matA(N,T):
    return(np.random.normal(0,1,(N,T)))

#Create the matrix B, defined as sqrt(Zeta)*A
def matB(N,T,c):
    A = matA(N,T)
    Zeta = matZeta(N,c)
    
    return(sqrtm(Zeta)@A)

#Create matrix C = 1/T BB^T
def matC(N,T,c):
    B = matB(N,T,c)
    return(1/T*B@ np.transpose(B))
    
#Create "num" matrices C and get all their eigenvalues
def main(N,T,c,num):
    values = []
    
    for j in range(num):
        C = matC(N,T,c)
        eigs = np.linalg.eigvals(C)
        for e in eigs:
            values.append(np.real(e))
    return(values)

def Dist(N,T,c,E):
    k = T/N
    E_p = (1/np.sqrt(k)+1)**2 * (1-c)
    E_m = (1/np.sqrt(k)-1)**2 * (1-c)
    
    return(k*np.sqrt((E_p - E)*(E-E_m))/(2*np.pi * E * (1-c)))

def plots(N,T,c,num):
    k = T/N
    E_p = (1/np.sqrt(k)+1)**2 * (1-c)
    E_m = (1/np.sqrt(k)-1)**2 * (1-c)

    xvals = np.linspace(E_m,E_p,100)

    yvals = [Dist(N,T,c,E) for E in xvals]

    
    arr, points, hist = plt.hist(main(N,T,c,50),bins = 100,range=(E_m-0.1,E_p+0.1),density=True,label="Eigenvalues")
    plt.plot(xvals,yvals,label="RMT")
    plt.title("Eigenvalue distribution")
