#Function that creates the real matrix of size n, random but symmetric entries, 
#each of which has mean 0 and variance 1 (2 in the case of the diagonal)
def exponential(n):
    v=np.random.exponential(1,(n,n))
    v = sqrt(0.5)*(v+np.transpose(v))
    return v

def uniform(n):
    v=np.random.uniform(0,1,(n,n))
    v = sqrt(0.5)*(v+np.transpose(v))
    return v

def normal(n):
    v = np.random.normal(0,1,(n,n))
    v = sqrt(0.5)*(v+np.transpose(v))
    return v

This function creates nmem matrices of dimensions ndxnd with a distribution given by "dist" (exponential, uniform or normal). Then, it calculates all the eigenvalues of such matrices, standarizes them and returns them in an array

#Returns a vector of dimension nd*nmem whose entries are nd eigenvalues for each of the nmem matrices
def main(nmem,nd,dist):
    xval = np.zeros(nmem*nd)
    j = 0
    for i in range(nmem):
        #create a matrix with the distribution dist:
        hmat = dist(nd)
        
        #get the eigenvalues, their mean and variance
        eigval = np.linalg.eigvals(hmat) 
        xm = sum(eigval)/len(eigval) 
        xv = np.std(eigval)
        
        #add to the array of eigenvalues the standarized eigenvalues we got.
        for k in range(nd):
            xval[j] = (eigval[k]-xm)/xv #standardizing the eigenvalues
            j = j + 1    

    return xval

def semicircle(x,R):  #Wigner's Semi-Circle Law. Create a semicircle with radius R
    return sqrt(R**2-x**2)*2/(np.pi*R**2)
#(I got the general expression for the law from wikipedia)
