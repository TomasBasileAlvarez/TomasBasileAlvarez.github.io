def euler_method(f, t0, tf, a0, r0,o0, tau,p1f):
    t_values = np.arange(t0, tf + h, h)
    steps = len(t_values)
    a_values = [a0]
    r_values = [r0]
    o_values = [o0]
    
    t = t0
    a = a0
    r = r0
    o = o0
    
    while t < tf:
        da,dr,do = f(t,a_values,r,o,a0,tau,steps,h,p1f)
        a = a + h *da
        r = r + h*dr
        o = o + h*do
        
        a_values.append(a)
        r_values.append(r)
        o_values.append(o)
        
        t += h
    return t_values, np.array(a_values), np.array(r_values), np.array(o_values)


def  f(t,a_values,r,o,a0,tau,steps,h,p1f):
#     print(p1(t))
    tau_ind = int(tau//h)
    da = p1f(t)/(1+p2*r*o) - p3*a_values[-1]
    dr = (o*r)**2/(p4+(o*r)**2) + p5 - p6*r
    
    if tau_ind > len(a_values):
        a_tau = a0
    else:
        a_tau = a_values[-tau_ind+1]
    
    do = a_tau - o
    
    return np.array([da, dr,do])
